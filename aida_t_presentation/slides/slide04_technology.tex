\textbf{\textcolor{aigreen}{1. Гибридная ходовая система:}}
\begin{itemize}
    \item Меканум-колеса для бетонных дорожек
    \item Рельсовые колеса для движения между рядами
    \item Активная амортизация для стабилизации
\end{itemize}

\textbf{\textcolor{aigreen}{2. Система технического зрения:}}
\begin{itemize}
    \item Регулируемая камерная мачта 0,1-3,0 м
    \item Промышленные Ethernet-камеры
    \item Многоракурсное сканирование
\end{itemize}

\textbf{\textcolor{aigreen}{3. ИИ-алгоритмы:}}
\begin{itemize}
    \item CNN для диагностики заболеваний
    \item RANSAC + PointNet для оценки объема
    \item Автономная навигация на ROS2
\end{itemize}

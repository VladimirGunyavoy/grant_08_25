\textbf{\large \textcolor{aiblue}{AIDA-T — комплексное решение для теплиц}}
\vspace{0.5cm}

\textbf{\textcolor{aigreen}{Ключевые возможности:}}
\begin{itemize}
    \item \textbf{Круглосуточный мониторинг} без участия человека
    \vspace{0.2cm}
    
    \item \textbf{Автономная навигация} по гибридной системе: бетон + рельсы
    \vspace{0.2cm}
    
    \item \textbf{ИИ-диагностика} заболеваний с точностью 86-87\%
    \vspace{0.2cm}
    
    \item \textbf{Оценка урожайности} в реальном времени
    \vspace{0.2cm}
    
    \item \textbf{Высокая точность} позиционирования ±1,5 мм на пиксель
\end{itemize}

\vspace{0.5cm}
\textcolor{aiorange}{\textbf{Ожидаемый эффект:}}
\begin{itemize}
    \item Снижение трудозатрат на 70\%
    \item Повышение урожайности на 15-20\%
    \item Окупаемость за 2,5-3 года
\end{itemize}
